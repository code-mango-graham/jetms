CREATE TABLE tbl_schoolyear (
    schoolyear_id INT AUTO_INCREMENT PRIMARY KEY,
    year_start YEAR NOT NULL,
    year_end YEAR NOT NULL,
    schoolyear_name VARCHAR(50) NOT NULL,
    status TINYINT(1) NOT NULL DEFAULT 1,
    schoolyear_remarks TINYINT(1) DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB;

CREATE TABLE tbl_office (
    office_id INT AUTO_INCREMENT PRIMARY KEY,
    office_name VARCHAR(100) NOT NULL,
    office_description TEXT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    office_remarks TINYINT(1) DEFAULT 1
) ENGINE=InnoDB;

CREATE TABLE tbl_level (
    level_id INT AUTO_INCREMENT PRIMARY KEY,
    level_name VARCHAR(100) NOT NULL,
    level_type VARCHAR(50) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    level_remarks TINYINT(1) DEFAULT 1
) ENGINE=InnoDB;

CREATE TABLE tbl_position ( 
    position_id INT AUTO_INCREMENT PRIMARY KEY, 
    position_title VARCHAR(100) NOT NULL, 
    description TEXT, 
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, 
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, 
    position_remarks TINYINT(1) NOT NULL DEFAULT 1 
)  ENGINE=InnoDB;

CREATE TABLE tbl_teacher (
    teacher_id INT AUTO_INCREMENT PRIMARY KEY,
    first_name VARCHAR(100) NOT NULL,
    middle_name VARCHAR(100),
    last_name VARCHAR(100) NOT NULL,
    extension_name VARCHAR(20),
    nick_name VARCHAR(100),
    gender ENUM('Male', 'Female', 'Other') NOT NULL,
    birthday DATE NOT NULL,
    phone_number VARCHAR(20),
    email VARCHAR(100) UNIQUE,
    position_id INT NOT NULL,
    department_id INT NOT NULL,
    date_hired date,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    teacher_remarks TINYINT(1) not null default 1
) ENGINE=InnoDB;
